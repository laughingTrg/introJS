var a = 1, b = 1, c, d; 
c = ++a; alert(c); //2 - так как, мы инкрементировали а на 1 потом вывели (1+1).
d = b++; alert(d); //1 - здесь изменен порядок операций сначала идет вывод текущего значения b (1), затем инкремент b на 1
c = (2+ ++a); alert(c); //5 - инкрементируем а на 1 (а =2+1= 3), итог 5
d = (2+ b++); alert(d); //4 - сначала складываем b (b =2+2= 4), выводим значение, затем снова инкрементируем b на 1 (2+1=3)
alert(a); //3 - теукщее значение а после двух инкрементов 
alert(b); //3 - теукщее значение b после двух инкрементов
